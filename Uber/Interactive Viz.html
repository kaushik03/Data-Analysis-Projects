df = df.rename(columns={"Date/Time": "Date_Time"})
df['Date_Time'] = pd.to_datetime(df['Date_Time'])
df["Day of Month"] = df["Date_Time"].dt.day
df["Day"] = df["Date_Time"].dt.day_name()
df["Hour"] = df['Date_Time'].dt.strftime('%H')
